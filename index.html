const campoSenha = document.getElementById("campo-senha"); const botaoGerar = document.getElementById("gerar"); const botaoCopiar = document.getElementById("copiar"); const mais = document.getElementById("mais"); const menos = document.getElementById("menos"); const valor = document.getElementById("valor"); const barra = document.getElementById("barra"); const textoForca = document.getElementById("texto-forca"); let tamanho = 12; const maiusculo = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"; const minusculo = "abcdefghijklmnopqrstuvwxyz"; const numeros = "0123456789"; const simbolos = "!@#$%&*()_+[]{}<>?/"; function gerarSenha() { let chars = ""; if (document.getElementById("maiusculo").checked) chars += maiusculo; if (document.getElementById("minusculo").checked) chars += minusculo; if (document.getElementById("numero").checked) chars += numeros; if (document.getElementById("simbolo").checked) chars += simbolos; if (chars === "") { alert("Selecione pelo menos uma opção!"); return; } let senha = ""; for (let i = 0; i < tamanho; i++) { const rand = Math.floor(Math.random() * chars.length); senha += chars[rand]; } campoSenha.value = senha; avaliarForca(senha); } function avaliarForca(senha) { let forca = 0; if (senha.length >= 8) forca++; if (senha.length >= 12) forca++; if (/[A-Z]/.test(senha)) forca++; if (/[0-9]/.test(senha)) forca++; if (/[^A-Za-z0-9]/.test(senha)) forca++; if (forca <= 2) { barra.style.width = "30%"; barra.style.background = "red"; textoForca.textContent = "Fraca"; } else if (forca <= 4) { barra.style.width = "70%"; barra.style.background = "orange"; textoForca.textContent = "Média"; } else { barra.style.width = "100%"; barra.style.background = "limegreen"; textoForca.textContent = "Forte"; } } botaoGerar.addEventListener("click", gerarSenha); botaoCopiar.addEventListener("click", () => { navigator.clipboard.writeText(campoSenha.value); alert("Senha copiada!"); }); mais.addEventListener("click", () => { tamanho++; valor.textContent = tamanho; }); menos.addEventListener("click", () => { if (tamanho > 4) { tamanho--; valor.textContent = tamanho; } });
